function GameRules:OnUpdateItemSpawns( deltaTime )

for _, itemSpawnId in ipairs(GameRules.itemSpawns) do

local itemSpawn = Shared.GetEntity( itemSpawnId )
if itemSpawn then

if itemSpawn:isa("ItemSpawn") and itemSpawn:GetCanRespawn(deltaTime) then
itemSpawn:SpawnItem()
end

end

end

end
